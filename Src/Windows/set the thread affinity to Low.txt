public static void setCurrentProgAffinity(String proc)
{
foreach (Process myCurrentProcess in Process.GetProcessesByName(proc))
{
myCurrentProcess.PriorityClass = System.Diagnostics.ProcessPriorityClass.Idle;
}
}
